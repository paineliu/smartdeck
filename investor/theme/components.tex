\newcommand{\SBCard}[3]{%
  \begin{beamercolorbox}[wd=#1,sep=0.15cm,rounded=true,shadow=false]{sbcard}%
    {\SBFontTitle\fontsize{11.2}{14}\selectfont\color{SBText}#2}\par
    \vspace{0.06cm}
    {\SBFontBody\fontsize{9.0}{12}\selectfont\color{SBMuted}#3}
  \end{beamercolorbox}%
}

\newcommand{\SBMetric}[3]{%
  \begin{beamercolorbox}[wd=#1,sep=0.12cm,center,rounded=true,shadow=false]{sbmetric}%
    {\SBFontTitle\fontsize{20}{24}\selectfont\color{SBPrimary}#2}\par
    \vspace{0.03cm}
    {\SBFontBody\fontsize{8.7}{11}\selectfont\color{SBMuted}#3}
  \end{beamercolorbox}%
}

\newcommand{\SBPill}[2]{%
  \colorbox{#1}{\strut\hspace{0.16cm}{\color{white}\SBFontTitle\fontsize{8.5}{11}\selectfont #2}\hspace{0.16cm}}%
}

\newcommand{\SBQuote}[1]{%
  \begin{beamercolorbox}[wd=0.86\textwidth,sep=0.20cm,center,rounded=true]{sbquote}%
    {\SBFontTitle\fontsize{17}{22}\selectfont\color{SBText}#1}
  \end{beamercolorbox}%
}

\newcommand{\SBThreeCards}[6]{%
  \begin{columns}[T,onlytextwidth]
    \column{0.31\textwidth}\SBCard{\linewidth}{#1}{#2}
    \column{0.31\textwidth}\SBCard{\linewidth}{#3}{#4}
    \column{0.31\textwidth}\SBCard{\linewidth}{#5}{#6}
  \end{columns}%
}

\newcommand{\SBFourMetrics}[8]{%

}
