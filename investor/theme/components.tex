\newcommand{\SBCard}[3]{%
  \begin{beamercolorbox}[wd=#1,ht=1.38cm,dp=0.30cm,sep=0.14cm,center,rounded=true,shadow=false]{sbcard}%
    {\SBFontTitle\fontsize{10.9}{13.0}\selectfont\color{SBText}#2}\par
    \vspace{0.07cm}
    {\SBFontBody\fontsize{8.7}{10.6}\selectfont\color{SBMuted}#3}
  \end{beamercolorbox}%
}

\newcommand{\SBMetric}[3]{%
  \begin{beamercolorbox}[wd=#1,ht=1.18cm,dp=0.25cm,sep=0.10cm,center,rounded=true,shadow=false]{sbmetric}%
    {\SBFontTitle\fontsize{19}{22}\selectfont\color{SBPrimary}#2}\par
    \vspace{0.03cm}
    {\SBFontBody\fontsize{8.4}{10.6}\selectfont\color{SBMuted}#3}
  \end{beamercolorbox}%
}

\newcommand{\SBPill}[2]{%
  \colorbox{#1}{\strut\hspace{0.16cm}{\color{white}\SBFontTitle\fontsize{8.5}{11}\selectfont #2}\hspace{0.16cm}}%
}

% A plain centered statement, without background box, for visual consistency.
\newcommand{\SBQuote}[1]{%
  {\SBLeadText{#1}}%
}

    \column{0.31\textwidth}\SBCard{\linewidth}{#3}{#4}
    \column{0.31\textwidth}\SBCard{\linewidth}{#5}{#6}
  \end{columns}%
}
