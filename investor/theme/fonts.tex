\defaultfontfeatures{Ligatures=TeX}

% Robust Latin font fallback chain for macOS, Linux and GitHub Actions.
\IfFontExistsTF{Inter}{
  \setmainfont{Inter}
  \setsansfont{Inter}
}{
  \IfFontExistsTF{Inter Display}{
    \setmainfont{Inter Display}
    \setsansfont{Inter Display}
  }{
    \IfFontExistsTF{Helvetica Neue}{
      \setmainfont{Helvetica Neue}
      \setsansfont{Helvetica Neue}
    }{
      \IfFontExistsTF{Noto Sans}{
        \setmainfont{Noto Sans}
        \setsansfont{Noto Sans}
      }{
        \IfFontExistsTF{TeX Gyre Heros}{
          \setmainfont{TeX Gyre Heros}
          \setsansfont{TeX Gyre Heros}
        }{
          \setmainfont{Lato}
          \setsansfont{Lato}
        }
      }
    }
  }
}

% Robust Chinese font fallback chain.

    \setCJKsansfont{Noto Sans CJK SC}
  }{
    \IfFontExistsTF{Source Han Sans SC}{
      \setCJKmainfont{Source Han Sans SC}
      \setCJKsansfont{Source Han Sans SC}
    }{
      \IfFontExistsTF{Songti SC}{
        \setCJKmainfont{Songti SC}
        \setCJKsansfont{Songti SC}
      }{
        \setCJKmainfont{FandolHei-Regular}
        \setCJKsansfont{FandolHei-Regular}
      }
    }
  }
}
