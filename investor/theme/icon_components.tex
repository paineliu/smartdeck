% v5.3 compile-safe icon layer.
% The visual token is a small rounded badge; SVG sources are stored in assets/icons
% and can replace this badge layer in v5.4 without changing page content.

\newcommand{\SBIconToken}[1]{%
  \begin{beamercolorbox}[wd=0.54cm,ht=0.34cm,dp=0.06cm,sep=0.02cm,center,rounded=true,shadow=false]{sbmetric}%
    {\SBFontTitle\fontsize{6.2}{7.2}\selectfont\color{SBPrimary}#1}
  \end{beamercolorbox}%
}

\newcommand{\SBIconFor}[1]{%
  \ifcsname SBIconName@#1\endcsname
    \SBIconToken{\csname SBIconName@#1\endcsname}%
  \else
    \SBIconToken{#1}%
  \fi
}

\newcommand{\SBIconCard}[4]{%

    \begin{minipage}[c][1.36cm][c]{0.20\linewidth}
      \centering
      \SBIconFor{#2}
    \end{minipage}%
    \begin{minipage}[c][1.36cm][c]{0.74\linewidth}
      {\SBFontTitle\fontsize{9.2}{11.0}\selectfont\color{SBText}#3}\par

}
