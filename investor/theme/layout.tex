\newcommand{\SBPageTitle}[1]{%
  \begin{tikzpicture}[remember picture,overlay]
    \node[anchor=north west, text=SBText, font=\sbTitleFont\fontsize{27}{31}\selectfont]
      at ([xshift=\SBLeft,yshift=-0.55cm]current page.north west) {#1};
    \draw[SBBlue,line width=1.4pt]
      ([xshift=\SBLeft,yshift=-1.16cm]current page.north west) --
      ([xshift=3.35cm,yshift=-1.16cm]current page.north west);
  \end{tikzpicture}
  \vspace{1.15cm}
}
